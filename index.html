Variables

En JavaScript las variables no tienen asociación de tipo en su definición.
El nombre de variable tiene que empezar por a - z , A - Z , $ , o _ . En su contenido puede tener estos además de números 0-9.
No se pueden utilizar palabras reservadas para nombres de variables ( for , in , if , etc)

  • var (JavaScript vanilla)
  • let (ECMAScript6)
  • const (ECMAScript6)

El valor de undefined se asigna a todas aquellas variables no inicializadas y también cuando se buscan propiedades de objetos no definidas en el mismo. En contexto de booleanos se considera false.

null

null es un valor que se ha declarado pero que que su contenido es nulo. En contexto de booleanos también se considera false.
Es un valor primitivo pero JavaScript al mirar su tipo lo considera object.

										console.log(null === undefined); // enforce type during check, displays false
										console.log(typeof null === 'object'); // true

Al haber coma flotante (flotating point) pueden haber errores en ciertas operaciones aritméticas con decimales por lo que no se recomienda hacer operaciones matemáticas complejas en JavaScript 

									
										alert(0.94 - 0.01);

number

Hay distintas maneras de representar un number en JavaScript 

						
							345; // un "integer", a pesar de que solo hay un tipo numérico en JavaScript
							34.5; // un número con coma flotante
							3.45e2; // otro número con coma flotante, que equivale a 345
							0b1011; //un binary integer que vale 11
							0o377; // un octal integer que equivale a 255
							0xFF; // un hexadecimal integer que equivale a 255, los digitos reprensentados por ...

							typeof NaN; // returns "number"

boolean

  • JavaScript tiene por booleano los literales true y false.
  • En contexto lógico tipo if, while, etc... por conversión de tipos son considerados booleanos:
    • false (falsy): 0, -0, null, NaN, undefined y '' (string de cadena vacia)
    • true (truthy): Todos los demás, incluyendo '0' y 'false' (estos como string)

Truthy y Falsy

						
							function testTruthyFalsy (val) { 
								return val ? console.log('truthy') : console.log('falsy'); 
							}
							 
							testTruthyFalsy(true);                   // truthy 
							testTruthyFalsy(false);                  // falsy 
							testTruthyFalsy(new Boolean(false));     // truthy (object is always true) 
							  
							testTruthyFalsy('');                     // falsy 
							testTruthyFalsy('Packt');                // truthy 
							testTruthyFalsy(new String(''));         // true (object is always true) 
							  
							testTruthyFalsy(1);                      // truthy 
							testTruthyFalsy(-1);                     // truthy 
							testTruthyFalsy(NaN);                    // falsy 
							testTruthyFalsy(new Number(NaN));        // truthy (object is always true) 
							  
							testTruthyFalsy({});                     // truthy (object is always true) 
							  
							var obj = { name: 'John' }; 
							testTruthyFalsy(obj);                    // truthy 
							testTruthyFalsy(obj.name);               // truthy

boolean: Conversiones de tipo automáticas al comparar (coercion)

A nivel interno, JavaScript realiza las siguientes conversiones de tipo automática en comparaciones

  • number and string: Intenta convertir el string a su valor númerico más cercano y luego lo redondea a su valor numérico más cercano
  • boolean: Si uno de los datos a comparar es boolean se convierte en 1 si es true o 0 si es false
  • object: Si un objeto es comparado con un string o number, el objeto se convierte en un string o number primitivo antes de compararse

boolean: Conversiones de tipo automáticas al comparar

									
									// Conversion automatica
									console.log(true == 2 ); // false... true → 1 !== 2 ← 2
									console.log(false == 2 ); // false... false → 0 !== 2 ← 2
									console.log(true == 1 ); // true.... true → 1 === 1 ← 1
									console.log(false == 0 ); // true.... false → 0 === 0 ← 0
									console.log(true == "2"); // false... true → 1 !== 2 ← "2"
									console.log(false == "2"); // false... false → 0 !== 2 ← "2"
									console.log(true == "1"); // true.... true → 1 === 1 ← "1"
									console.log(false == "0"); // true.... false → 0 === 0 ← "0"
									console.log(false == "" ); // true.... false → 0 === 0 ← ""
									console.log(false == NaN); // false... false → 0 !== NaN
			
									console.log(NaN == NaN); // false...... NaN es el único no equivalente a nada, incluso NaN.
			
									// comparación con tipo (no hay conversión de tipos y valores)
									console.log(true === 1); // false...... los tipos no coinciden
			
									// Forzando conversión de tipo
									console.log(true === !!2); // true.... los tipos y valores coinciden
									console.log(true === !!0); // false... los tipos coinciden pero los valores no
									console.log( 1 ? true : false); // true.... solo ±0 y NaN son “falsy” numbers
									console.log("0" ? true : false); // true.... en string solo carácter vacio es “falsy”
									console.log(Boolean({})); // true.... todos los objetos son “truthy”

symbol

Nuevo en ECMAScript6. Un Symbol es un identificador único e inmutable. 

						
							x=Symbol(1);
							y=Symbol(1);
							x==y; // false
							var arr=[x,y];
							arr[x]=1;
							arr[y]=2; // x and y are unique keys for the array arr
							arr[x]; // displays 1
							arr[y]; // displays 2
							x=Symbol(3);
							arr; // displays [Symbol(1),Symbol(1)]
							arr[x]; // is now undefined
							x=Symbol(1);
							arr[x]; // undefined

while vs do while

La principal diferencia consiste en que "while" no llega a ejecutar el bloque de código si la condición no se cumple y "do while" lo ejecuta pero no repite iteración si no se cumple.

while vs do while

						
							var numOfCustomers = 10;
							var textCustomer = 'Texto de while: \n';
							while (numOfCustomers > 0) {
								textCustomer += 'Hay ' +  numOfCustomers + ' cliente/s \n';
								numOfCustomers = numOfCustomers - 1;
							}
							textCustomer += 'Texto de do while: \n';
							// versus:
							numOfCustomers = 10;
							do {
								textCustomer += 'Hay ' +  numOfCustomers + ' cliente/s \n';
								numOfCustomers = numOfCustomers - 1;
							} while (numOfCustomers > 0);
							alert(textCustomer);

while vs do while

						
							var numOfCustomers = 10;
							var textCustomer = 'Texto de while: \n';
							while (false) {
								textCustomer += 'Hay ' +  numOfCustomers + ' cliente/s \n';
								numOfCustomers = numOfCustomers - 1;
							}
							textCustomer += 'Texto de do while: \n';
							// versus:
							numOfCustomers = 10;
							do {
								textCustomer += 'Hay ' +  numOfCustomers + ' cliente/s \n';
								numOfCustomers = numOfCustomers - 1;
							} while (false);
							alert(textCustomer);

Funciones

Una función es una porción de código que puede ser llamada y que cada vez que se hace ejecuta el código en su interior.

Conviene usarlas para evitar duplicidad de código de tareas repetitivas que se puedan abstraer.

Funciones

Las funciones pueden llamarse mientras estén definidas en el mismo scope

						
							printAmount();
							function printAmount() {
								alert(amount ? amount.toFixed(2) : 'Aún no está definido amount');
							}
							var amount = 99.99;
							printAmount(); // 99.99
							
							amount = amount * 2;
							printAmount(); // 199.98

Funciones con parámetros

Se les pueden pasar parámtros a las funciones, incluso otras funciones 

						
							var TAX_RATE = 0.08;
							function calculateFinalPurchaseAmount(amt) {
								// calcula el montante con los impuestos
								amt = amt + (amt * TAX_RATE);
								// regresa el nuevo montante
								return amt;
								// tb seria valido return (amt + (amt * TAX_RATE));
							}
							var amount = 99.99;
							amount = calculateFinalPurchaseAmount(amount);
							alert( amount.toFixed( 2 ) ); // 107.99

Funciones con función como parámetros

						
							var TAX_RATE = 0.08;
							function calculateFinalPurchaseAmount(amt, functionAmt) {
								// Ejecuamos la función que se le ha pasado como parámetro
								return functionAmt(amt);
							}
							var amount = 99.99;
							amount = calculateFinalPurchaseAmount(amount, function(amt) {
								return (amt + (amt * TAX_RATE));
							});
							alert( amount.toFixed( 2 ) ); // 107.99

Hoisting

Proceso por el cual definiciones de funciones y variables son llevadas a la parte superior de ejecución del scope donde se encuentran. 

						
							var a = 2;
							foo(); // works because `foo()`
							// declaration is "hoisted"
							function foo() {
								a = 3;
								console.log( a ); // 3
								var a; // declaration is "hoisted"
								// to the top of `foo()`
							}
							console.log( a ); // 2

Whitespace and semicolons

Dos principales problemas:

  • Cinco carácteres que pueden ser principio de sentencia o extensión: ( [ / + -
  • Cinco comandos restrictivos: return | throw | break | continue | ++ o --

5 carácteres de inicio o extensión. Ej:

						
						a = b + c
						(d + e).foo()

						// Será ejecutado como:
						// a = b + c(d + e).foo();

						a = b + c
						;(d + e).foo()
						
						// Será ejecutado como:
						// a = b + c;
						// (d + e).foo();

5 comandos restrictivos. Ej:

						
							return
							a + b;

							// Devuelve undefined. Será ejecutado como:
							// 	  return;
							// 	  a + b;
							// Debería ser reescrito como:
							//    return a + b;
